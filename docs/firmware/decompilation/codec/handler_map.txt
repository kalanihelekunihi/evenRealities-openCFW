# ===========================================================================
# 1. LVP APPLICATION LIFECYCLE HANDLERS
#    Module-based suspend/resume lifecycle (bare-metal, no RTOS)
# ===========================================================================

_LvpAppSuspend (string at 0x0004a9b4, also 0x00013390)
  Trigger: System entering low-power state (IDLE mode entry)
  Action: Suspends the LVP core framework — stops audio capture pipeline,
          halts NPU inference, saves state for resume
  Lifecycle: Called before IDLE mode entry
  Log: Printed as suspend/resume function name for debugging

_LvpAppResume (string at 0x0004a9c4, also 0x0001339f)
  Trigger: System exiting low-power state (wakeup event)
  Action: Resumes LVP core — restarts audio capture, reinitializes NPU,
          restores audio pipeline state
  Lifecycle: Called after wakeup from IDLE mode

SampleAppSuspend (string at 0x0004a9e4, also 0x000133b8, 0x0004aaac)
  Trigger: Audio output application entering suspend
  Action: Suspends the I2S output application (SampleApp = I2S output driver)
  Log tag: [I2S_OUT_APP] ---- %s ---- (0x0004a9f8)
  Note: "SampleApp" is the NationalChip SDK name for the I2S output application
        module, NOT a demo — it is the production audio output path

SampleAppResume (string at 0x0004a9d4, also 0x000133c9, 0x0004aac0)
  Trigger: Audio output application exiting suspend
  Action: Resumes I2S output path — re-enables DAC/I2S clocking, restores
          audio buffer state
  Log tag: [I2S_OUT_APP] ---- %s ----

# 2. UART MESSAGE HANDLERS
#    Communication with Apollo510b main SoC via TinyFrame serial protocol
# ===========================================================================

UartMessageAsyncSuspend (string at 0x0004a534, also 0x0001332e)
  Trigger: System entering low-power mode
  Action: Suspends UART message handler — stops polling for messages from
          the Apollo510b host SoC, disasserts VC_MESSAGE notify pin
  Note: "Async" indicates interrupt-driven (not polled) message handling

UartMessageAsyncResume (string at 0x0004a54c, also 0x00013346)
  Trigger: System resuming from low-power mode
  Action: Resumes UART message handler — re-enables interrupt-driven RX,
          reasserts VC_MESSAGE notify pin for host communication

_uartRecvCallback (string at 0x0004aad0)
  Trigger: UART byte received (interrupt-driven)
  Action: Processes incoming bytes from Apollo510b via TinyFrame protocol
  Protocol: TinyFrame with CRC32 (separate instance from inter-eye TinyFrame)
  Note: This is the ISR-level byte receiver that feeds the TinyFrame decoder

_uartMsgInit (string at 0x0004aae4)
  Trigger: Boot initialization
  Action: Initializes UART message subsystem — configures UART peripheral,
          sets up TinyFrame instance, registers command handlers
  Configures: VC_MESSAGE UART port and NOTIFY GPIO pin

receive_gsensor_test_cmd (string at 0x0004ab14: "receive gsensor test cmd")

# ===========================================================================
# 3. AUDIO OUTPUT CALLBACKS
#    gx_audio_out_* API callback handlers
# ===========================================================================

gx_audio_out_config_cb (string at 0x00049bb4)
  Trigger: Audio output subsystem configured with callback
  Action: Registers a callback function for audio output events (buffer
          underrun, playback complete, etc.)
  Used by: SampleApp (I2S output application)

# The audio output API (20 functions) uses callbacks for asynchronous
# notification of audio events. The callback is set via gx_audio_out_config_cb
# and invoked by the DMA completion handler when audio frames are consumed.
#
# Audio output lifecycle:
#   gx_audio_out_alloc_playback → gx_audio_out_config_buffer →
#   gx_audio_out_config_pcm → gx_audio_out_config_i2s →
#   gx_audio_out_config_dac → gx_audio_out_config_cb →
#   gx_audio_out_push_frame (loop) → gx_audio_out_drain_frame →
#   gx_audio_out_exit
#
# Suspend/resume:
#   gx_audio_out_suspend (0x00049c88) / gx_audio_out_resume (0x00049ca0)

# ===========================================================================
# 5. BEAMFORMING MODE SWITCH HANDLER
#    GSC/IMCRA/raw mode switching
# ===========================================================================

switch_bf_cmd_handler (string at 0x0001340e: "switch bf cmd")
  Trigger: Host sends SwitchBFMode command OR CLI "switch bf cmd"
  Action: Switches between 3 denoise modes:
    MODE 1 — DOUBLE_MIC_DENOISE: GSC adaptive beamforming (dual-mic)
      Init log: [LVP_MODE_DENOISE][DOUBLE_MIC_DENOISE]Init gsc Beamforming
    MODE 2 — SINGLE_MIC_DENOISE: IMCRA noise estimation (single-mic)
      Init log: [LVP_MODE_DENOISE][SINGLE_MIC_DENOISE]Init Imcra
    MODE 3 — ORIGIN_MIC: Raw microphone passthrough (no processing)
      Init log: [LVP_MODE_DENOISE][ORIGIN_MIC]Change origin mic
  Cross-ref: [codec.host]SwitchBFMode on Apollo510b side
  Error: [codec.host]SwitchBFMode failed: %d, response status: 0x%02X

switch_wakeup_cmd_handler (string at 0x0004ac34: "switch wakeup cmd")
  Trigger: Host sends SwitchWakeupMode command OR CLI "switch wakeup cmd"
  Action: Configures wakeup source / wake word detection mode
  Wake words: "hey_even" (0x00017254), "hi_even" (0x000172ac)
  Cross-ref: [codec.host]SwitchWakeupMode on Apollo510b side

# 7. DMA INTERRUPT HANDLERS
#    DMA controller for audio data transfer
# ===========================================================================

# DMA interrupt handlers are inferred from FIFO interrupt strings:

dma_tx_fifo_empty_handler (inferred from "Transmit FIFO empty interrupt." at 0x00007fbc)
  Trigger: DMA transmit FIFO becomes empty
  Action: Refill transmit FIFO from audio buffer or signal underrun

dma_tx_fifo_overflow_handler (inferred from "Transmit FIFO overflow interrupt." at 0x00007fdc)
  Trigger: DMA transmit FIFO overflow
  Action: Error recovery — discard excess data, log error

dma_rx_fifo_underflow_handler (inferred from "Receive FIFO underflow interrupt." at 0x00008000)
  Trigger: DMA receive FIFO underflow (read from empty FIFO)
  Action: Error recovery

dma_rx_fifo_overflow_handler (inferred from "Receive FIFO overflow interrupt." at 0x00008024)
  Trigger: DMA receive FIFO overflow
  Action: Error recovery — data loss

dma_rx_fifo_full_handler (inferred from "Receive FIFO full interrupt." at 0x00008048)
  Trigger: DMA receive FIFO full
  Action: Drain FIFO to audio processing pipeline

dma_multimaster_contention_handler (inferred from "multi-master contention interrupt." at 0x00008068)
  Trigger: Multi-master bus contention on SPI bus
  Action: Arbitration / retry

dma_xip_rx_overflow_handler (inferred from "xip receive fifo overflow interrupt." at 0x0000808c)
  Trigger: XIP (eXecute In Place) receive FIFO overflow during flash read
  Action: Error recovery — XIP read stall

  Trigger: OTP (One-Time Programmable) region read request

  Error: "read flash otp error" on access failure

# 10. CPU EXCEPTION HANDLER
#     C-SKY CK803S exception vector
# ===========================================================================

cpu_exception_handler (string at 0x000082bc: "CPU Exception : %u")
  Trigger: Unhandled CPU exception (invalid instruction, bus fault, etc.)
  Action: Dumps register state and halts:
    vr%d: %08x  (0x000082d0) — vector registers
    r%d: %08x   (0x000082dc) — general registers
    epsr: %8x   (0x000082e8) — exception PSR (C-SKY status register)
    epc : %8x   (0x000082f4) — exception PC (program counter at fault)
  Also: Stack overflow detection:
    [LVP]Error, The Stack May Used Up, Top[%#x],[Addr:%#x, Value:%#x] (0x0004a4a8)
  Assert: %s:%u: %s: Assertion `%s' failed. (0x00008ff4)

# 11. IDLE MODE HANDLERS
#     Power state management
# ===========================================================================

idle_mode_init_handler (string at 0x0004a598: "[IDLE]Init IDLE mode")
  Trigger: System entering IDLE power state
  Action: Initializes IDLE mode — suspends non-essential modules,
          reduces clock frequencies, enables wakeup sources

idle_mode_exit_handler (string at 0x0004a580: "[IDLE]Exit IDLE mode")
  Trigger: Wakeup event detected (VAD, G-sensor, UART, GPIO)
  Action: Exits IDLE mode — resumes modules, restores clocks,
          re-enables full audio pipeline


# ===========================================================================
# 12. DENOISE MODE LIFECYCLE HANDLERS
#     [LVP_MODE_DENOISE] module init/exit
# ===========================================================================

denoise_init_handler (string at 0x0004a890: "[LVP_MODE_DENOISE]Init denoise mode")
  Trigger: Audio pipeline startup or denoise mode switch
  Action: Initializes selected denoise mode (GSC, IMCRA, or raw)
  Sequence:
    1. LvpAudioInInit — initialize audio input
    2. _beamformingInit or _imarcInit — init selected algorithm
    3. FrameDrcStateInit — init DRC post-processor
    4. Denoise Init OK!!!
  Errors: BF malloc g_bf_mictmp failed, ImcraStateInit Fail!, etc.

denoise_exit_handler (string at 0x0004a5c4: "[LVP_MODE_DENOISE]Exit denoise mode")
  Trigger: Audio pipeline shutdown or mode switch
  Action: Tears down current denoise mode, frees buffers
